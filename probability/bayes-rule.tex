\section{Bayes Rule}
\begin{slide}[Bayes rule]
\begin{shaded}
\begin{equation}
  P(A|B) = P(B|A){P(A) \over P(B)}
\end{equation}
\end{shaded}
\end{slide}
\begin{slide}[Law of total probability]
\vspace{-1cm}
\begin{shaded}
\begin{equation}
  P(B) = P(B|A_1)P(A_1)+\ldots P(B|A_n)P(A_n)
\end{equation}
\end{shaded}
where $\{A_i\}_{i=1}^n$ are partitions of sample space
\end{slide}
