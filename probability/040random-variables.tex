  \begin{quote}
  Random variable is a function from the sample space to the real line.
  \end{quote}
  \begin{align*}
    S &= \{HHH, HHT, HTH , HTT, THH, THT, TTH,  TTT\}\\
    X &= \text{is a random variable representing no of heads}\\
    &X(HHH) = 3\hspace{20pt} 
    X(HHT) = 2 \hspace{20pt}
    X(HTH) = 2\hspace{20pt}
    X(HTT) = 1\\
    &X(THH) = 2\hspace{20pt}
    X(THT) = 1\hspace{20pt}
    X(TTH) = 1\hspace{20pt}
    X(TTT) = 0\\
   \end{align*}
     $$P(X=2) = 3/8$$
