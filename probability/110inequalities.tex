\begin{slide}[Cauchy Schwartz]

	$$E(XY) \leq \sqrt{E(X^2)E(Y^2)}$$

if i$X$, $Y$ is uncorrelated, $E(XY) = E(X)E(Y)$

\end{slide}

\end{slide}
